Documentation

Minimal commands to reproduce the workflow

# 1) HIL recording
lerobot-human-inloop-record \
  --robot.type=so100_follower \
  --teleop.type=so100_leader \
  --dataset.repo_id=local/evo_demo

# 2) Dataset report
lerobot-dataset-report --dataset local/evo_demo --json

# 3) Value train
lerobot-value-train \
  --dataset.repo_id=local/evo_demo \
  --value.type=pistar06 \
  --steps=2000

# 4) Value infer + ACP labels
lerobot-value-infer \
  --dataset.repo_id=local/evo_demo \
  --inference.checkpoint_path=outputs/value_train/evo_value_demo \
  --acp.enable=true
